  desc = "granada_civil_war.1.desc"
  picture = CIVIL_WAR_eventPicture

  is_triggered_only = yes
  immediate = {
    hidden_effect = {
      if = {
        limit = {

          }

        random_owned_province = {
          limit = {
            is_core = ROOT
            is_capital = no

            religion = ROOT
            size = 2
            leader = "优素福 纳斯尔"  # Yûsuf Nasrid
            win = yes
          }
          add_local_autonomy = 25
        }
      }
      else = {
        capital_scope = {
          spawn_rebels = {
            type = pretender_rebels
            religion = ROOT
            size = 2
            leader = "优素福 纳斯尔"  # Yûsuf Nasrid
          }
        }
      }
      set_country_flag = had_granada_succession_war
    }
  }
  option = {
    name = "granada_civil_war.1.a"
    add_prestige = -10
  }
}
